# Leetcode ：79. 单词搜索

## 一、题目链接
# [LeetCode 79. 单词搜索](https://leetcode.cn/problems/word-search/description/?envType=study-plan-v2&envId=top-100-liked)

## 二、题目描述
在字母矩阵中找是否有指定的字符串


## 三、解题思路与代码实现
### 1. 核心思路
使用回溯算法，比较简单，当下标等于字符串的长度时说明找到了，返回True，否则搜索完没找到就返回False


### 2. 代码实现
#### 回溯算法：
```C++

class Solution {
    vector<pair<int, int>> dirs = {{0, 1}, {1, 0}, {-1, 0}, {0, -1}};
    vector<vector<bool>> visited;
    bool answer = false;
    void backtrack(vector<vector<char>>& board, string word, int index,
                   int cur_i, int cur_j) {
        if (index == word.size()) {
            answer = true;
            return ;
        }
        if (cur_i < 0 || cur_i >= board.size() || cur_j < 0 ||
            cur_j >= board[0].size() || visited[cur_i][cur_j] ||
            board[cur_i][cur_j] != word[index])
            return ;

        visited[cur_i][cur_j] = true;
        for (auto dir : dirs) {
            backtrack(board, word, index + 1, cur_i + dir.first,
                      cur_j + dir.second);
        }
        visited[cur_i][cur_j] = false;
        return ;
    }

public:
    bool exist(vector<vector<char>>& board, string word) {
        if (word.empty())
            return true;
        int m = board.size();
        int n = board[0].size();
        answer = false;
        visited = vector<vector<bool>>(m, vector<bool>(n, false));
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (board[i][j] == word[0]) {
                    backtrack(board, word, 0, i, j);
                    if (answer)
                        return true;
                }
            }
        }
        return answer;
    }
};
```
